## 方法二：自动提取高频词

### 2.1 原理

使用**正则表达式**从标题中提取连续的汉字词组（2-8字），统计词频，过滤停用词。

### 2.2 代码实现

```python
import json
import re
from collections import Counter

def extract_high_freq_words(data, min_length=2, min_freq=3):
    """
    从新闻标题中自动提取高频词

    Args:
        data: 新闻JSON数据
        min_length: 最小词长度（默认2字）
        min_freq: 最小出现次数（默认3次）

    Returns:
        高频词列表 [(词, 频次), ...]
    """
    # 1. 提取所有标题
    titles = [news['title'] for news in data['news']]
    all_text = ' '.join(titles)

    # 2. 停用词表（常见无意义词）
    stopwords = {
        # 助词
        '的', '了', '是', '在', '和', '与', '或', '等', '都', '也',
        '这', '那', '有', '被', '把', '让', '给', '到', '从', '为',
        # 代词
        '一个', '什么', '怎么', '如何', '为什么', '哪些', '可以',
        '他们', '我们', '你们', '自己', '这个', '那个', '这些', '那些',
        # 副词/连词
        '可能', '应该', '需要', '已经', '正在', '还将', '还是', '只是',
        '但是', '因为', '所以', '如果', '虽然', '不是', '没有', '不会',
        '不能', '不要', '还有', '就是', '也是', '都是', '就在',
        # 时间词
        '小时', '年前', '年后', '分钟', '昨天', '今天', '明天',
        # 媒体词
        '作者', '来源', '记者', '编辑', '报道', '文章', '内容', '标题',
    }

    # 3. 使用正则提取中文词组（连续2-8个汉字）
    words = re.findall(r'[\u4e00-\u9fa5]{2,8}', all_text)

    # 4. 过滤停用词
    filtered_words = [w for w in words
                      if w not in stopwords
                      and len(w) >= min_length]

    # 5. 统计词频
    word_freq = Counter(filtered_words)

    # 6. 过滤低频词
    result = [(word, count) for word, count in word_freq.items()
              if count >= min_freq]

    # 7. 按频率排序
    result.sort(key=lambda x: x[1], reverse=True)

    return result
```

### 2.3 使用示例

```python
# 读取数据
with open('news_raw.json', 'r', encoding='utf-8') as f:
    data = json.load(f)

# 提取高频词
high_freq_words = extract_high_freq_words(data, min_length=2, min_freq=5)

# 输出TOP30
print("自动发现的高频词 TOP30:")
for word, count in high_freq_words[:30]:
    print(f"  {word}: {count}")
```

### 2.5 优点与局限

**优点**：
- ✅ 无需预设词典，自动发现新热点
- ✅ 实现简单，无需第三方库
- ✅ 可发现意料之外的热词

**局限**：
- ⚠️ 可能提取无意义词（如"亿元"、"之后"）
- ⚠️ 分词不精确（"人形机器人"可能被拆成多个词）
- ⚠️ 无法区分实体类型（不知道是公司还是人名）

```python
# -*- coding: utf-8 -*-
"""
新闻关键词分析工具
方法二 + 方法三 组合使用
"""

import json
import re
from collections import Counter
from pathlib import Path


class NewsKeywordAnalyzer:
    """新闻关键词分析器"""

    # 停用词表
    STOPWORDS = {
        '的', '了', '是', '在', '和', '与', '或', '等', '都', '也',
        '这', '那', '有', '被', '把', '让', '给', '到', '从', '为',
        '一个', '什么', '怎么', '如何', '为什么', '哪些', '可以',
        '他们', '我们', '你们', '自己', '这个', '那个', '这些', '那些',
        '可能', '应该', '需要', '已经', '正在', '还将', '还是', '只是',
        '但是', '因为', '所以', '如果', '虽然', '不是', '没有', '不会',
        '不能', '不要', '还有', '就是', '也是', '都是', '就在',
        '小时', '年前', '年后', '分钟', '昨天', '今天', '明天',
    }

    # 实体词典
    ENTITY_DICT = {
        '公司': {
            '苹果', '华为', '小米', 'OPPO', 'vivo', '三星',
            '特斯拉', '比亚迪', '蔚来', '理想', '小鹏', '吉利',
            '阿里', '阿里巴巴', '腾讯', '字节', '字节跳动', '美团', '百度',
            '京东', '拼多多', '网易', '滴滴', '快手', 'B站', '微博',
            'OpenAI', 'Anthropic', 'Claude', 'Google', '微软', 'Meta',
            '茅台', '五粮液', '宁德时代', '中芯国际', '大疆',
        },
        '人物': {
            '张雪', '董宇辉', '特朗普', '马斯克', '雷军', '罗永浩',
            '周鸿祎', '俞敏洪', '李佳琦', '薇娅', '罗翔', '张一鸣',
            '马云', '马化腾', '刘强东', '王兴', '黄峥', '李想', '李斌',
        },
        '地点': {
            '美国', '中国', '伊朗', '巴基斯坦', '以色列', '俄罗斯', '乌克兰',
            '日本', '韩国', '印度', '德国', '英国', '法国',
            '北京', '上海', '深圳', '广州', '杭州', '武汉', '成都', '重庆',
            '欧洲', '中东', '东南亚',
        },
        '行业': {
            'AI', '人工智能', '新能源', '电动车', '芯片', '半导体',
            '互联网', '电商', '直播', '外卖', '游戏', '电影',
            '房地产', '医疗', '教育', '金融',
            '无人驾驶', '自动驾驶', '萝卜快跑', 'ChatGPT', '大模型',
        },
    }

    def __init__(self, data_file):
        """初始化分析器"""
        self.data_file = Path(data_file)
        self.data = None
        self.titles = None

    def load_data(self):
        """加载数据"""
        with open(self.data_file, 'r', encoding='utf-8') as f:
            self.data = json.load(f)
        self.titles = [news['title'] for news in self.data['news']]
        return self

    def extract_high_freq_words(self, min_length=2, min_freq=3, top_n=50):
        """
        方法二：自动提取高频词

        Returns:
            [(词, 频次), ...]
        """
        all_text = ' '.join(self.titles)

        # 正则提取中文词组
        words = re.findall(r'[\u4e00-\u9fa5]{2,8}', all_text)

        # 过滤
        filtered_words = [w for w in words
                          if w not in self.STOPWORDS
                          and len(w) >= min_length]

        # 统计
        word_freq = Counter(filtered_words)

        # 返回TOP N
        return word_freq.most_common(top_n)

    def extract_entities(self, top_n=10):
        """
        方法三：实体识别分类

        Returns:
            {类别: [(实体, 频次), ...]}
        """
        all_text = ' '.join(self.titles)
        result = {}

        for category, entities in self.ENTITY_DICT.items():
            counter = Counter()
            for entity in entities:
                count = all_text.count(entity)
                if count > 0:
                    counter[entity] = count
            result[category] = counter.most_common(top_n)

        return result

    def find_new_entities(self, high_freq_words, top_n=20):
        """
        发现新实体（高频词中不在词典里的词）

        Returns:
            [(词, 频次), ...] 新发现的实体
        """
        # 合并所有已知实体
        all_known_entities = set()
        for entities in self.ENTITY_DICT.values():
            all_known_entities.update(entities)

        # 找出高频词中不在词典的
        new_entities = [(word, count) for word, count in high_freq_words
                        if word not in all_known_entities
                        and word not in self.STOPWORDS]

        return new_entities[:top_n]

    def analyze(self):
        """执行完整分析"""
        print("=" * 60)
        print("新闻关键词分析报告")
        print("=" * 60)
        print(f"总新闻数: {self.data['total']}")
        print(f"来源分布: {self.data['by_source']}")
        print()

        # 方法二：高频词
        print("=" * 60)
        print("方法二：自动提取高频词")
        print("=" * 60)
        high_freq_words = self.extract_high_freq_words()
        print("\n高频词 TOP30:")
        for word, count in high_freq_words[:30]:
            print(f"  {word}: {count}")

        # 方法三：实体分类
        print()
        print("=" * 60)
        print("方法三：实体识别分类")
        print("=" * 60)
        entities = self.extract_entities()
        for category, items in entities.items():
            if items:
                print(f"\n【{category}】TOP10:")
                for name, count in items:
                    print(f"  {name}: {count}")

        # 发现新实体
        print()
        print("=" * 60)
        print("新发现实体（需要关注）")
        print("=" * 60)
        new_entities = self.find_new_entities(high_freq_words)
        print("\n这些高频词不在词典中，可能是新热点:")
        for word, count in new_entities:
            print(f"  {word}: {count}")

        return {
            'high_freq_words': high_freq_words,
            'entities': entities,
            'new_entities': new_entities,
        }


# 使用示例
if __name__ == '__main__':
    analyzer = NewsKeywordAnalyzer('news_raw.json')
    analyzer.load_data()
    result = analyzer.analyze()
```

## 方法对比总结

| 维度 | 方法二（自动提取） | 方法三（实体识别） |
|------|-------------------|-------------------|
| **核心原理** | 正则提取中文词组 | 预设词典匹配 |
| **依赖** | 无需预设词典 | 需要预设词典 |
| **发现性** | ✅ 强，可发现新热点 | ❌ 弱，只能识别已知实体 |
| **精确性** | ⚠️ 一般，可能有无意义词 | ✅ 高，精确分类 |
| **分类能力** | ❌ 无 | ✅ 有（公司/人物/地点/行业） |
| **维护成本** | ✅ 低，只需维护停用词 | ⚠️ 中，需要持续更新词典 |
| **推荐场景** | 探索性分析，发现新热点 | 精确统计，生成报告 |

**最佳实践**：组合使用
1. 先用**方法二**自动提取高频词，发现新热点
2. 再用**方法三**对已知实体进行分类统计
3. 最后将**新发现的实体**加入词典，持续优化

```python
def identify_entity_with_llm(word, context_titles):
    """
    使用大模型识别实体类型

    Args:
        word: 待识别的词
        context_titles: 包含该词的标题列表（提供上下文）

    Returns:
```
