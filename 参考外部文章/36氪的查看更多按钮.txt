def human_scroll(page, times=30):
    """
    人类式滚动 + 【新增】自动点击“查看更多”按钮
    """
    prev_count = 0
    no_new_count = 0
    
    for i in range(times):
        # 随机鼠标移动
        for _ in range(random.randint(3, 6)):
            page.mouse.move(random.randint(100, 1400), random.randint(100, 800), steps=random.randint(8, 15))
            time.sleep(random.uniform(0.1, 0.4))
        
        # 大段滚动
        scroll_distance = random.randint(800, 1500)
        page.evaluate(f"window.scrollBy(0, {scroll_distance})")
        time.sleep(random.uniform(2.0, 5.0))
        
        # 【新增】检测并点击“查看更多”按钮（核心修复）
        if page.locator('text=查看更多').count() > 0:
            print("🔄 发现‘查看更多’按钮，正在点击加载下一批...")
            page.locator('text=查看更多').first.click()
            time.sleep(random.uniform(3.0, 5.0))   # 等待新内容加载
            debug_captcha(page, f"点击查看更多后_第{i+1}次")
        
        # 每 3 次检查一次验证码
        if (i + 1) % 3 == 0:
            solve_slider(page)
        
        # 统计文章数
        try:
            article_count = len(page.query_selector_all('.information-flow-item'))
            if article_count == 0:
                article_count = len(page.query_selector_all('a[href*="/p/"]'))
            
            if article_count > prev_count:
                no_new_count = 0
                prev_count = article_count
                print(f"  第 {i+1} 次滚动完成，当前文章数：{article_count} (+{article_count - prev_count})")
            else:
                no_new_count += 1
                print(f"  第 {i+1} 次滚动完成，当前文章数：{article_count} (无新内容 {no_new_count}次)")
                
                if no_new_count >= 5:
                    print("  尝试刷新加载...")
                    page.evaluate("window.scrollTo(0, 0)")
                    time.sleep(2.0)
                    page.evaluate("window.scrollTo(0, document.body.scrollHeight)")
                    time.sleep(3.0)
                    no_new_count = 0
        except:
            pass
        
        # 诊断（每 5 次）
        if (i + 1) % 5 == 0:
            debug_captcha(page, f"滚动第{i+1}次_后")
