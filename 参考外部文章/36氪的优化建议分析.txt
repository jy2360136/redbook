from playwright.sync_api import sync_playwright
import cv2
import numpy as np
import time
import random
import base64
import re
import json
from pathlib import Path

def human_drag(page, start_x, start_y, distance):
    page.mouse.move(start_x, start_y)
    page.mouse.down()
    steps = random.randint(28, 38)
    for i in range(steps):
        ratio = i / steps
        offset = distance * (ratio ** 1.85) + random.uniform(-4.5, 4.5)
        y_offset = random.uniform(-3, 3)
        page.mouse.move(start_x + offset, start_y + y_offset, steps=1)
        time.sleep(random.uniform(0.007, 0.023))
    page.mouse.up()
    time.sleep(random.uniform(0.6, 1.4))

def detect_gap(bg_base64: str, piece_base64: str) -> int:
    bg_data = base64.b64decode(re.sub('^data:image/[^;]+;base64,', '', bg_base64))
    piece_data = base64.b64decode(re.sub('^data:image/[^;]+;base64,', '', piece_base64))
    
    bg = cv2.imdecode(np.frombuffer(bg_data, np.uint8), cv2.IMREAD_COLOR)
    piece = cv2.imdecode(np.frombuffer(piece_data, np.uint8), cv2.IMREAD_COLOR)
    
    bg_gray = cv2.cvtColor(bg, cv2.COLOR_BGR2GRAY)
    piece_gray = cv2.cvtColor(piece, cv2.COLOR_BGR2GRAY)
    
    bg_edge = cv2.Canny(bg_gray, 100, 200)
    piece_edge = cv2.Canny(piece_gray, 100, 200)
    
    result = cv2.matchTemplate(bg_edge, piece_edge, cv2.TM_CCOEFF_NORMED)
    _, _, _, max_loc = cv2.minMaxLoc(result)
    gap = max_loc[0] * 0.98   # 微调比例，根据实际图片调整
    return int(gap)

def solve_slider(page):
    try:
        page.wait_for_selector('text=按住左边按钮拖动完成上方拼图', timeout=6000)
        print("🔒 检测到滑块验证码，开始自动处理...")
    except:
        return True  # 没有验证码

    # === 提取图片（36氪常见 selector，根据实际弹出后 F12 微调）===
    bg_selector = "img.basicImg, .geetest_canvas_fullbg img, canvas, img[src*='basic']"
    piece_selector = "img.actionImg, .geetest_canvas_slice img, canvas, img[src*='slice']"
    
    bg_img = page.locator(bg_selector).first
    piece_img = page.locator(piece_selector).first
    
    bg_base64 = bg_img.evaluate("el => el.src || el.toDataURL('image/png')")
    piece_base64 = piece_img.evaluate("el => el.src || el.toDataURL('image/png')")
    
    distance = detect_gap(bg_base64, piece_base64)
    print(f"📏 计算缺口距离: {distance}px")
    
    # 拖动滑块
    slider_btn = page.locator('.geetest_slider_button, button[aria-label*="拖动"], [class*="slider"]')
    box = slider_btn.bounding_box()
    human_drag(page, box['x'] + 12, box['y'] + 12, distance)
    
    time.sleep(1.8)
    if not page.locator('text=按住左边按钮拖动完成上方拼图').is_visible():
        print("✅ 滑块验证通过！")
        return True
    else:
        print("❌ 验证失败，点击刷新重试...")
        page.locator('text=刷新').click()
        time.sleep(1)
        return False

# ==================== 主程序 ====================
with sync_playwright() as p:
    browser = p.chromium.launch(
        headless=False,
        args=['--disable-blink-features=AutomationControlled', '--no-sandbox']
    )
    
    # 持久化 Context（推荐）
    state_path = Path("36kr_state.json")
    if state_path.exists():
        context = browser.new_context(storage_state=json.loads(state_path.read_text(encoding="utf-8")))
    else:
        context = browser.new_context(
            viewport={"width": 1440, "height": 900},
            user_agent="Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/134.0.0.0 Safari/537.36",
            locale="zh-CN",
            timezone_id="Asia/Shanghai"
        )
    
    # Stealth 注入
    page = context.new_page()
    page.add_init_script("""
        Object.defineProperty(navigator, 'webdriver', {get: () => undefined});
        Object.defineProperty(navigator, 'plugins', {get: () => [1,2,3,4,5]});
        Object.defineProperty(navigator, 'languages', {get: () => ['zh-CN', 'zh']});
    """)
    
    page.goto("https://36kr.com/information/web_news/", wait_until="networkidle")
    time.sleep(random.uniform(2.5, 4.5))
    
    max_retries = 3
    for attempt in range(max_retries):
        if solve_slider(page):
            break
        if attempt == max_retries - 1:
            print("⚠️ 多次验证失败，请手动处理或检查网络")
    
    # ==================== 这里写你的爬取逻辑 ====================
    print("🚀 开始提取新闻列表...")
    news_list = page.locator('article, .news-item, [class*="article"]').all()
    for news in news_list[:10]:  # 示例：提取前10条
        title = news.locator('h2, h3, a').first.inner_text(timeout=2000).strip()
        link = news.locator('a').first.get_attribute('href')
        print(f"标题: {title}\n链接: {link}\n{'-'*60}")
    
    # 保存 state
    context.storage_state(path=str(state_path))
    
    # 保持浏览器打开方便调试
    input("按 Enter 关闭浏览器...")
    browser.close()
