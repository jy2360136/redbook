def debug_captcha(page, step_name: str):
    """自动诊断 36氪验证码状态（关键！）"""

def human_drag(page, start_x, start_y, distance):
    """人类拖动轨迹（贝塞尔 + 随机抖动 + 加速度）"""
    page.mouse.move(start_x, start_y)
    page.mouse.down()
    steps = random.randint(28, 38)
    for i in range(steps):
        ratio = i / steps
        offset = distance * (ratio ** 1.85) + random.uniform(-4.5, 4.5)
        y_offset = random.uniform(-3, 3)
        page.mouse.move(start_x + offset, start_y + y_offset, steps=1)
        time.sleep(random.uniform(0.007, 0.023))
    page.mouse.up()
    time.sleep(random.uniform(0.6, 1.4))

def detect_gap(bg_base64: str, piece_base64: str) -> int:
    """OpenCV 识别滑块缺口"""
    bg_data = base64.b64decode(re.sub('^data:image/[^;]+;base64,', '', bg_base64))
    piece_data = base64.b64decode(re.sub('^data:image/[^;]+;base64,', '', piece_base64))
    bg = cv2.imdecode(np.frombuffer(bg_data, np.uint8), cv2.IMREAD_COLOR)
    piece = cv2.imdecode(np.frombuffer(piece_data, np.uint8), cv2.IMREAD_COLOR)
    bg_gray = cv2.cvtColor(bg, cv2.COLOR_BGR2GRAY)
    piece_gray = cv2.cvtColor(piece, cv2.COLOR_BGR2GRAY)
    bg_edge = cv2.Canny(bg_gray, 100, 200)
    piece_edge = cv2.Canny(piece_gray, 100, 200)
    result = cv2.matchTemplate(bg_edge, piece_edge, cv2.TM_CCOEFF_NORMED)
    _, _, _, max_loc = cv2.minMaxLoc(result)
    return int(max_loc[0] * 0.98)

def solve_slider(page):
    """字节 iframe 滑块验证码自动破解（核心改进）"""
    try:
        page.wait_for_selector('#captcha_container', timeout=5000)
        print("🔒 检测到字节滑块验证码（iframe）...")
    except:
        return True

    frame_locator = page.frame_locator('iframe[src*="rmc.bytedance.com/verifycenter/captcha"]')

    # selector（可根据 debug 截图微调）
    bg_selector = "canvas#bg-canvas, canvas.bg, img[alt*='background'], .captcha-bg"
    piece_selector = "canvas#piece-canvas, canvas.slice, img[alt*='piece'], .captcha-piece"
    slider_btn_selector = "div[class*='slide'], .verify-slide-btn, button[aria-label*='拖动'], .captcha-slider-thumb"

    bg_element = frame_locator.locator(bg_selector).first
    piece_element = frame_locator.locator(piece_selector).first
    slider_btn = frame_locator.locator(slider_btn_selector).first

    bg_base64 = bg_element.evaluate("el => el.tagName === 'CANVAS' ? el.toDataURL('image/png') : (el.src || el.toDataURL('image/png'))")
    piece_base64 = piece_element.evaluate("el => el.tagName === 'CANVAS' ? el.toDataURL('image/png') : (el.src || el.toDataURL('image/png'))")

    distance = detect_gap(bg_base64, piece_base64)
    print(f"📏 计算缺口距离: {distance}px")

    box = slider_btn.bounding_box()
    human_drag(page, box['x'] + 15, box['y'] + 15, distance)

    time.sleep(1.8)
    if not page.locator('#captcha_container').is_visible():
        print("✅ 滑块验证通过！")
        return True
    else:
        print("❌ 验证失败，点击刷新...")
        page.locator('text=刷新').click()
        return False

def human_scroll(page, times=8):
    """人类式滚动（防触发验证码）"""
    for i in range(times):
        for _ in range(random.randint(3, 6)):
            page.mouse.move(random.randint(100, 1400), random.randint(100, 800), steps=random.randint(8, 15))
            time.sleep(random.uniform(0.1, 0.4))
        scroll_distance = random.randint(300, 700)
        page.evaluate(f"window.scrollBy(0, {scroll_distance})")
        time.sleep(random.uniform(1.5, 4.0))
        if (i + 1) % 2 == 0:
            solve_slider(page)
        debug_captcha(page, f"滚动第{i+1}次_后")
        print(f"第 {i+1} 次人类滚动完成，当前文章数: {len(page.locator('a[href^=\"/p/\"]').all())}")
